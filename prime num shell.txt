#!/bin/bash

echo "Enter lower limit:"
read low
echo "Enter upper limit:"
read high

for ((i=low; i<=high; i++))
do
    flag=1
    for ((j=2; j*j<=i; j++))
    do
        if (( i % j == 0 )); then
            flag=0; break
        fi
    done
    (( i > 1 && flag == 1 )) && echo $i
done
